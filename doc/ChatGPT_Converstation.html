python
import sys
from PyQt5.QtWidgets import QApplication, QFileSystemModel, QTreeView, QWidget, QVBoxLayout


class DirTreeView(QWidget):
    def __init__(self, dir_path):
        super().__init__()

        self.setWindowTitle("Directory Tree Viewer")

        # Create a file system model
        self.model = QFileSystemModel()
        self.model.setRootPath(dir_path)

        # Create a tree view widget
        self.tree_view = QTreeView()
        self.tree_view.setModel(self.model)
        self.tree_view.setRootIndex(self.model.index(dir_path))

        # Add the tree view to the main layout
        layout = QVBoxLayout()
        layout.addWidget(self.tree_view)
        self.setLayout(layout)


if __name__ == '__main__':
    app = QApplication(sys.argv)

    # Specify the directory path to display
    dir_path = '/path/to/directory'

    # Create the directory tree view widget
    tree_view = DirTreeView(dir_path)
    tree_view.show()

    sys.exit(app.exec_())

This script creates a DirTreeView widget that displays the directory tree of the specified directory. The QFileSystemModel class is used to create a model of the file system that can be used to populate the tree view widget. The QTreeView widget is used to display the directory tree.

To use the script, replace '/path/to/directory' with the path to the directory you want to display and run the script. A window should appear displaying the directory tree.

import sys
from PyQt5.QtCore import QDir
from PyQt5.QtWidgets import QApplication, QFileSystemModel, QTreeView, QWidget, QVBoxLayout, QHBoxLayout, QFileDialog, QLabel, QPushButton, QCheckBox


class DirTreeView(QWidget):
    def __init__(self):
        super().__init__()

        # Create a file system model
        self.model = QFileSystemModel()

        # Only display directories in the tree view by default
        self.filter_directories_only = True
        self.update_filter()

        # Create a tree view widget
        self.tree_view = QTreeView()

        # Create a label to display the selected directory path
        self.path_label = QLabel("Selected Path: ")

        # Create a button to select a directory
        self.select_dir_button = QPushButton("Select Directory")

        self.show_files_checkbox = QCheckBox("Show Files")
        self.show_files_checkbox.setChecked(not self.filter_directories_only)
        self.show_files_checkbox.stateChanged.connect(self.update_filter)

        # Create a button to expand all folders
        self.expand_button = QPushButton("Expand All")
        self.expand_button.clicked.connect(self.expand_all)

        # Create a button to collapse all folders

        self.tree_view.setRootIndex(self.model.index(''))

        # Update the filter to show or hide files based on the checkbox state
        self.filter_directories_only = not self.show_files_checkbox.isChecked()
